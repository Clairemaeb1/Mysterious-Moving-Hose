        <script>
            document.addEventListener('DOMContentLoaded', () => {
                const checklistItems = document.querySelectorAll('.checklist-item');
                checklistItems.forEach(item => {
                    item.addEventListener('click', () => {
                        item.classList.toggle('completed');
                    });
                });

                const missionReportForm = document.getElementById('missionReportForm');
                const submissionMessage = document.getElementById('submissionMessage');

                missionReportForm.addEventListener('submit', (event) => {
                    event.preventDefault(); // Stop the form from really sending

                    const screenshotUrl = encodeURIComponent(document.getElementById('screenshotUrl').value);
                    const findings = encodeURIComponent(document.getElementById('findings').value);
                    const recipientEmail = 'Clairemaeb1@gmail.com';
                    const subject = encodeURIComponent('G.P.P.P. Mission Report: Moving Hose Mystery');
                    const body = encodeURIComponent(`Agent Findings Report:\n\nScreenshot Link: ${decodeURIComponent(screenshotUrl)}\n\nWhat I Found: ${decodeURIComponent(findings)}\n\n--- End of Report ---`);

                    if (screenshotUrl.trim() === '' || findings.trim() === '') {
                        submissionMessage.textContent = 'üö® Report Missing Stuff: Please add your screenshot link AND what you found! üö®';
                        submissionMessage.classList.remove('hidden', 'text-green-600');
                        submissionMessage.classList.add('text-red-600');
                    } else {
                        const mailtoLink = `mailto:${recipientEmail}?subject=${subject}&body=${body}`;
                        window.open(mailtoLink, '_blank'); // Opens your email app in a new tab

                        submissionMessage.textContent = 'üöÄ Mission Report Ready! Just hit "Send" in your email app. Great job, Agent!';
                        submissionMessage.classList.remove('hidden', 'text-red-600');
                        submissionMessage.classList.add('text-green-600');
                        missionReportForm.reset(); // Clears the boxes after you send it
                    }
                });
            });
        </script>
